## 🎯 设计目标

1. **主机发现**：局域网内自动发现其他 GeniusClaw 主机
2. **服务注册**：主机启动时自动注册服务
3. **双向通信**：主机之间可靠的数据同步
4. **零配置**：用户无需手动配置，开箱即用

## 📡 发现协议

### 1. mDNS 服务发布

```dart
// Dart/mDNS 服务发布示例
class MdnsService {
  static const String serviceType = '_geniusclaw._tcp';
  static const int servicePort = 8443;

  // 启动服务发布
  Future<void> startAdvertising(GeniusClawDevice device) async {
    final txtRecord = {
      'version': '1.0',
      'device_id': device.id,
      'role': device.role,
      'capabilities': device.capabilities.join(','),
      'hostname': device.hostname,
    };

    await _zeroconf.publish(
      serviceType: serviceType,
      port: servicePort,
      txtRecord: txtRecord,
      domain: 'local.',
    );
  }
}
```

### 2. 服务发现与查询

```dart
// 服务发现
class ServiceDiscovery {
  // 监听新设备加入
  Stream<List<GeniusClawDevice>> watchDevices() async* {
    await for (final result in _zeroconf.browse(serviceType)) {
      final devices = await _resolveDevices(result);
      yield devices;
    }
  }

  // 解析设备详情
  Future<GeniusClawDevice> _resolveDevice(BrowseResult result) async {
    final serviceInfo = await _zeroconf.resolve(result);
    
    return GeniusClawDevice(
      id: serviceInfo.txtRecord['device_id'],
      name: result.name,
      ip: serviceInfo.ip,
      port: serviceInfo.port,
      role: serviceInfo.txtRecord['role'],
      capabilities: serviceInfo.txtRecord['capabilities'].split(','),
    );
  }
}
```

## 📡 通信协议

### gRPC 服务定义

```protobuf
// geniusclaw.proto
syntax = "proto3";

package geniusclaw;

service GeniusClawService {
  // 设备注册与心跳
  rpc Register(RegisterRequest) returns (RegisterResponse);
  rpc Heartbeat(HeartbeatRequest) returns (HeartbeatResponse);
  
  // 设备状态同步
  rpc SyncStatus(StatusRequest) returns (stream StatusUpdate);
  
  // 技能同步
  rpc SyncSkills(SyncSkillsRequest) returns (SyncSkillsResponse);
  
  // AI 对话转发
  rpc ForwardChat(ChatRequest) returns (stream ChatResponse);
  
  // 文件传输
  rpc TransferFile(stream FileChunk) returns (TransferResponse);
}

// 设备注册
message RegisterRequest {
  string device_id = 1;
  string hostname = 2;
  string role = 3;
  repeated string capabilities = 4;
  NetworkInfo network = 5;
}

message RegisterResponse {
  bool success = 1;
  string message = 2;
  repeated PeerInfo peers = 3;
}

// 心跳
message HeartbeatRequest {
  string device_id = 1;
  int64 timestamp = 2;
  DeviceStatus status = 3;
}

message HeartbeatResponse {
  bool ok = 1;
}

// 设备状态
message StatusUpdate {
  string device_id = 1;
  DeviceStatus status = 2;
  int64 timestamp = 3;
}

message DeviceStatus {
  bool online = 1;
  int32 cpu_usage = 2;
  int32 memory_usage = 3;
  int32 active_skills = 4;
  string ai_model = 5;
}

// 网络信息
message NetworkInfo {
  string ip_address = 1;
  string mac_address = 2;
  string hostname = 3;
}

// 对等节点信息
message PeerInfo {
  string device_id = 1;
  string hostname = 2;
  string ip_address = 3;
  int32 port = 4;
}
```

### USB 连接实现

```dart
// USB 连接管理
class USBConnectionManager {
  // 监听 USB 设备连接
  Stream<USBDevice> get onDeviceConnected {
    return _usbSerial.onConnect.map((port) => USBDevice(
      name: port.name,
      pid: port.productId,
      vid: port.vendorId,
    ));
  }

  // 初始化串口通信
  Future<void> initialize(String portPath) async {
    final port = await _usbSerial.open(portPath);
    
    // 配置串口参数
    await port.configure(
      baudRate: 115200,
      dataBits: 8,
      stopBits: 1,
      parity: Parity.none,
    );

    // 读取设备信息
    final info = await _readDeviceInfo(port);
    
    // 通知上层
    _controller.add(USBDeviceConnected(info));
  }

  // 读取设备信息
  Future<DeviceInfo> _readDeviceInfo(SerialPort port) async {
    // 发送识别命令
    port.write([0x01, 0x02, 0x03, 0x04]); // CMD_GET_INFO
    
    // 等待响应
    final response = await port.read(64);
    
    // 解析响应
    return DeviceInfo.parse(response);
  }
}
```

## 🔒 安全机制

### 1. 设备认证

```dart
// 设备认证
class DeviceAuth {
  // 生成设备密钥
  String generateDeviceKey(String deviceId) {
    final salt = getRandomSalt();
    final key = pbkdf2(deviceId, salt, iterations: 10000);
    return '$salt:$key';
  }

  // 验证设备
  Future<bool> verifyDevice(String deviceId, String key) async {
    final storedKey = await _getStoredKey(deviceId);
    return constantTimeCompare(key, storedKey);
  }
}
```
